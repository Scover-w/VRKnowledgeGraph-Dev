149 private async Task WaitRepositoryConnection()
150 {
151 bool isConnectedToRepo = false;
152 bool isFirstTest = true;
153
154 while (!isConnectedToRepo)
155 {
156
157 if (!isFirstTest)
158 {
159 await Task.Run(() =>
160 {
161 Thread.Sleep(5000);
162 });
163
164 }
165
166 var status = await _graphDbAPI.DoRepositoryExist();
167 isFirstTest = false;
168
169 switch (status)
170 {
171 case RepositoryStatus.Exist:
172 isConnectedToRepo = true;
173 break;
174 case RepositoryStatus.ExistButUnreadable:
175 _loadingBarUI.Refresh(0f, "Le dépôt ne possède pas les autorisations de lecture requises pour communiquer avec celui-ci. Veuillez modifier vos droits ou changer d'identifants.");
176 break;
177 case RepositoryStatus.Nonexistent:
178 _loadingBarUI.Refresh(0f, "Le dépôt n'existe pas dans la base de données.");
179 break;
180 case RepositoryStatus.Unauthorized:
181 _loadingBarUI.Refresh(0f, "Connection au dépôt non autorisé. Veuillez modifier vos droits ou changer d'identifants.");
182 break;
183 case RepositoryStatus.CouldntConnect:
184 _loadingBarUI.Refresh(0f, "Impossible de se connecter à la base de donnée. Veuillez vous assurer que le serveur est en ligne.");
185 break;
186 case RepositoryStatus.Failed:
187 _loadingBarUI.Refresh(0f, "Connexion échouée.");
188 break;
189 }

223 private void AddLogInUnityThread(string logMessage)
224 {
225 var newLine = Environment.NewLine;
226 _logSb.Insert(0, logMessage + newLine);
227
228 string txt = _logSb.ToString();
229
230 string[] lines = txt.Split(new[] { newLine }, StringSplitOptions.None);
231 if (lines.Length > 12)
232 {
233 int lineLength = lines[lines.Length - 2].Length;
234 int lastLineIndex = txt.LastIndexOf(newLine) - lineLength;
235 if (lastLineIndex >= 0)
236 {
237 _logSb.Remove(lastLineIndex, lineLength);
238 }
239 }
240
241 _logTxt.text = _logSb.ToString();
242 }
