155 private string CreatePrefix(string namespce)
156 {
157 var prefix = namespce.Replace("http://", "")
158 .Replace("https://", "");
159
160 prefix = prefix.Split("/")[0];
161
162 prefix = prefix.Replace("www.", "")
163 .Replace(".com", "")
164 .Replace(".fr", "")
165 .Replace(".org", "");
166
167 prefix = CapitalizeOnCharacter(prefix, '.');

169 prefix = CapitalizeOnCharacter(prefix, '_');

174 private void AddPrefix(string namespce, string prefix)
175 {
176
177 if(namespce.Contains("ark:"))
178 {
179 _namespacesAndPrefixs.Add(namespce, prefix);
180 return;
181 }
182
183
184 HashSet<string> existingPrefixs = new();
185
186 foreach(string existingPrefix in _namespacesAndPrefixs.Values)
187 {
188 existingPrefixs.Add(existingPrefix);
189 }
190
191 if (!existingPrefixs.Contains(prefix))
192 {
193 _namespacesAndPrefixs.Add(namespce, prefix);
194 return;
195 }
196
197 int i = 1;
198 string prefixNumbered;
199
200 do
201 {
202 i++;
203 prefixNumbered = prefix + i;
204
205
206 } while (existingPrefixs.Contains(prefixNumbered));
207
208 _namespacesAndPrefixs.Add(namespce, prefixNumbered);

212 public string CapitalizeOnCharacter(string input, char separator)
213 {
214 string[] words = input.Split(separator);
215
216 for (int i = 1; i < words.Length; i++)
217 {
218 words[i] = CultureInfo.CurrentCulture.TextInfo.ToTitleCase(words[i]);
219 }
220
221 return string.Concat(words);

250 public async Task CreateOntologyTrees(GraphDBAPI graphDBAPI, DataSynchroManager dataSynchroManager)
251 {
252 var sparqlQuery = "PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> " +
253 "PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#> " +
254 "PREFIX owl: <http://www.w3.org/2002/07/owl#> " +
255 "SELECT ?s ?p ?o " +
256 "FROM <http://ontology> " +
257 "WHERE { { " +
258 "?s ?p ?o. " +
259 "FILTER( ( (?p = rdf:type && (?o = rdfs:Class || ?o = owl:Class)) || ?p = rdfs:subClassOf ) ) }}";
260
261 dataSynchroManager.AddLog("Récupération des noeuds ontologiques dans GraphDb");
262 var json = await graphDBAPI.SelectQuery(sparqlQuery);
263 var data = await JsonConvertHelper.DeserializeObjectAsync<JObject>(json);
264 _ontoTreeDict = new();
265
266 dataSynchroManager.AddLog("Création des arbres ontologiques");
267
268 foreach (JToken binding in data["results"]["bindings"])
269 {
270 string sValue = binding["s"]["value"].Value<string>();
271 string pValue = binding["p"]["value"].Value<string>();
272 string oValue = binding["o"]["value"].Value<string>();
273
274
275 if(pValue == "http://www.w3.org/1999/02/22-rdf-syntax-ns#type")
276 {
277 var namescpe = sValue.ExtractUri().namespce;
278
279 if (!namescpe.StartsWith("http"))
280 continue;
281
282 var ontologyTree = TryGetOrCreateOntologyTree(namescpe);
283 OntoNode ontoNode = new(sValue);
284 ontologyTree.AddOntoNode(ontoNode);
285
286 continue;
287 }
288
289
290 if (pValue == "http://www.w3.org/2000/01/rdf-schema#subClassOf")
291 {
292 var nameSpceA = sValue.ExtractUri().namespce;
293 var nameSpceB = oValue.ExtractUri().namespce;
294
295 if(nameSpceA != nameSpceB)
296 {
297 continue;
298 }
299
300 if (!nameSpceA.StartsWith("http"))
301 continue;
302
303 var ontologyTree = TryGetOrCreateOntologyTree(nameSpceA);
304
305 OntoNode sOntoNode = new(sValue);
306 OntoNode oOntoNode = new(oValue);
307
308 sOntoNode = ontologyTree.TryGetOrCreateOntoNode(sOntoNode);
309 oOntoNode = ontologyTree.TryGetOrCreateOntoNode(oOntoNode);
310
311 sOntoNode.AddOntoNodeSource(oOntoNode);
312 oOntoNode.AddOntoNodeTarget(sOntoNode);
313 continue;
314 }
315 }
316 foreach (var ontologyTree in _ontoTreeDict.Values)
317 {
318 ontologyTree.SetRootAndDepth();
319 }

364 private void LoadDefaultPrefixsList(TextAsset textAsset)
365 {
366 _defaultPrefixsDict = new();
367
368 using StreamReader reader = new(new MemoryStream(textAsset.bytes));
369 string line;
370
371 while ((line = reader.ReadLine()) != null)
372 {
373
374 line = line.Replace("@prefix ", "");
375
376 var elements = line.Split("<");
377
378 string prefix = elements[0];
379 string namespce = elements[1];
380
381
382 prefix = prefix.Replace(" ", "").Replace(":", "");
383 namespce = namespce.Replace(">", "");
384 namespce = namespce.Substring(0, namespce.Length - 1);
385
386 if (_defaultPrefixsDict.ContainsKey(namespce))
387 continue;
388
389 _defaultPrefixsDict.Add(namespce, prefix);
390 }
