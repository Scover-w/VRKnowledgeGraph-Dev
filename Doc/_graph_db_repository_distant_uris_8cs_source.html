192 private bool ExtractName(string xmlContent, out string property, out string value)
193 {
194 property = "";
195 value = "";
196
197
198 try
199 {
200 using StringReader stringReader = new(xmlContent);
201 using XmlReader xmlReader = XmlReader.Create(stringReader);
202
203 while (xmlReader.Read())
204 {
205 if (xmlReader.NodeType != XmlNodeType.Element)
206 continue;
207
208 var name = xmlReader.Name.ToLower();
209
210 if (!(name.Contains("label") || name.Contains("title") || name.Contains("name")))
211 continue;
212
213 property = xmlReader.Name;
214 value = xmlReader.ReadElementContentAsString();
215
216 return true;
217 }
218 }
219 catch(Exception)
220 {
221 return false;
222 }
223
224
225 return false;
226 }
