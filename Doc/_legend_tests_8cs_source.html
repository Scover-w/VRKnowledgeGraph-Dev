179public class GaussianKDEEstimator
180{
181 public float BandwidthH = .5f;
182
183 public float MinDensity { get; private set; }
184 public float MaxDensity { get; private set; }
185
186
187 public List<float> GaussianKDE(List<float> data, int resolution)
188 {
189 // Initialize
190 List<float> kdeValues = new List<float>();
191 float min = data.Min();
192 float max = data.Max();
193 float range = max - min;
194 float step = range / (resolution - 1);
195
196 MinDensity = float.MaxValue;
197 MaxDensity = float.MinValue;
198
199 int n = data.Count; // Number of data points
200
201 // Calculate KDE values for the range
202 for (int i = 0; i < resolution; i++)
203 {
204 float x = min + i * step;
205 float sum = 0.0f;
206
207 foreach (float xi in data)
208 {
209 float z = (x - xi) / BandwidthH;
210 sum += (float)(Math.Exp(-0.5 * z * z) / Math.Sqrt(2 * Math.PI));
211 }
212
213 float density = sum / (n * BandwidthH);
214 kdeValues.Add(density);
215
216 if (density < MinDensity)
217 MinDensity = density;
218 else if(density > MaxDensity)
219 MaxDensity = density;
220
221
222 }
223
224 return kdeValues;
225 }
