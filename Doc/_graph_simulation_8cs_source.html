135 private bool CalculateNodeSimuData(NodgesSimuData nodgesSimuData)
136 {
137 var nodesSimuData = nodgesSimuData.NodeSimuDatas;
138 var edgesSimuData = nodgesSimuData.EdgeSimuDatas;
139
140
141 var config = _graphConfiguration.SimuParameters;
142
143 float coulombDistance = config.CoulombDistance;
144 float springDistance = config.SpringDistance;
145 float stopVelocity = config.StopVelocity;
146
147 float coulombForce = config.CoulombForce;
148 float springForce = config.SpringForce;
149 float damping = config.Damping;
150 float maxVelocity = config.MaxVelocity;
151 float invMaxVelocity = 1f / config.MaxVelocity;
152
153 float velocitySum = 0f;
154 float tickDeltaTime = config.TickDeltaTime;
155
156 // Apply the repulsion force between all nodes
157
158 // Coulomb Force, Apply the repulsion force between all nodes
159 foreach (var idAndNodeDataA in nodesSimuData)
160 {
161 var nodeDataA = idAndNodeDataA.Value;
162 nodeDataA.Velocity = Vector3.zero;
163
164 foreach (var idAndNodeDataB in nodesSimuData)
165 {
166 var nodeDataB = idAndNodeDataB.Value;
167 if (nodeDataA.UID == nodeDataB.UID)
168 continue;
169
170 Vector3 direction = nodeDataB.Position - nodeDataA.Position;
171 float distance = direction.magnitude;
172
173 if (distance > coulombDistance)
174 continue;
175
176 float repulsiveForce = coulombForce * (1f / distance);
177
178 Vector3 forceVector = repulsiveForce * direction.normalized;
179 nodeDataA.Velocity -= forceVector;
180 }
181
182 }
183
184 // Spring force - Attractive Force between connected nodes
185 foreach (var idAndEdgeData in edgesSimuData)
186 {
187 var edge = idAndEdgeData.Value;
188
189 if (!nodesSimuData.TryGetValue(edge.UidA, out NodeSimuData nodeDataA))
190 continue;
191
192 if (!nodesSimuData.TryGetValue(edge.UidB, out NodeSimuData nodeDataB))
193 continue;
194
195
196
197 Vector3 direction = nodeDataB.Position - nodeDataA.Position;
198
199 float distance = direction.magnitude;
200
201 if (distance < springDistance)
202 continue;
203
204 float attractiveForce = (distance - springDistance) * springForce;
205 Vector3 forceVector = attractiveForce * direction.normalized;
206 nodeDataA.Velocity += forceVector;
207 nodeDataB.Velocity -= forceVector;
208 }
209
210
211 // Apply velocity
212 foreach (var idAndNodeData in nodesSimuData)
213 {
214 var nodeData = idAndNodeData.Value;
215 nodeData.Velocity *= damping;
216
217 if (nodeData.Velocity.magnitude > maxVelocity)
218 nodeData.Velocity /= nodeData.Velocity.magnitude * invMaxVelocity;
219
220 velocitySum += nodeData.Velocity.magnitude;
221 nodeData.Position += nodeData.Velocity * tickDeltaTime;
222 }
223
224 float velocityGraph = velocitySum / (float)nodesSimuData.Count;
225
226 return velocityGraph < stopVelocity;
227 }

230 private float GetMaxRadius(NodgesSimuData nodgesSimuData)
231 {
232 float maxSqrDistance = 0f;
233
234 var nodesSimuData = nodgesSimuData.NodeSimuDatas;
235
236 foreach (var nodeSimuData in nodesSimuData.Values)
237 {
238 var sqrDistance = nodeSimuData.Position.sqrMagnitude;
239
240 if (sqrDistance > maxSqrDistance)
241 maxSqrDistance = sqrDistance;
242 }
243
244 return Mathf.Sqrt(maxSqrDistance);
245 }
